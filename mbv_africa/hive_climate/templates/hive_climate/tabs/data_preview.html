Sample Hive Queries

Analytical queries for climate intelligence

Monthly Average Temperature by Region

SELECT region, month, 
       AVG(temp_mean) as avg_temp,
       AVG(precipitation) as avg_precip
FROM africa_climate_observations
WHERE year = 2024
GROUP BY region, month
ORDER BY region, month;

Coastal Ocean Temperature Anomalies

SELECT station_name, observation_date,
       sea_surface_temp,
       sea_surface_temp - AVG(sea_surface_temp) OVER (
         PARTITION BY station_id 
         ORDER BY observation_date 
         ROWS BETWEEN 365 PRECEDING AND 1 PRECEDING
       ) as sst_anomaly
FROM africa_climate_observations
WHERE sea_surface_temp IS NOT NULL
  AND year >= 2020;
