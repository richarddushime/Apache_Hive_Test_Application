Sample Hive Queries

Analytical queries for climate intelligence on the africa_climate_observations table

üîç Monthly Average Temperature by Region

SELECT region, month, 
       ROUND(AVG(temp_mean), 2) as avg_temp,
       ROUND(AVG(precipitation), 2) as avg_precip,
       COUNT(*) as observation_count
FROM africa_climate_observations
WHERE year = 2024 AND temp_mean IS NOT NULL
GROUP BY region, month
ORDER BY region, month;

üåä Coastal Ocean Temperature Anomalies (Window Function)

SELECT station_name, observation_date,
       sea_surface_temp,
       ROUND(sea_surface_temp - AVG(sea_surface_temp) OVER (
         PARTITION BY station_id 
         ORDER BY observation_date 
         ROWS BETWEEN 365 PRECEDING AND 1 PRECEDING
       ), 2) as sst_anomaly
FROM africa_climate_observations
WHERE sea_surface_temp IS NOT NULL
  AND year >= 2020
ORDER BY sst_anomaly DESC
LIMIT 100;

üìä Temperature Extremes by Country

SELECT country,
       MAX(temp_max) as hottest_recorded,
       MIN(temp_min) as coldest_recorded,
       ROUND(AVG(temp_mean), 1) as avg_temperature
FROM africa_climate_observations
WHERE year BETWEEN 2020 AND 2024
GROUP BY country
ORDER BY hottest_recorded DESC;
