CREATE TABLE statement with optimizations

CREATE TABLE africa_climate_observations (
  station_id STRING,
  station_name STRING,
  country STRING,
  latitude DOUBLE,
  longitude DOUBLE,
  observation_date DATE,
  month INT,
  temp_max DOUBLE,
  temp_min DOUBLE,
  temp_mean DOUBLE,
  precipitation DOUBLE,
  humidity DOUBLE,
  sea_surface_temp DOUBLE,
  ocean_salinity DOUBLE
)
PARTITIONED BY (year INT, region STRING)
CLUSTERED BY (station_id) INTO 32 BUCKETS
STORED AS ORC
TBLPROPERTIES (
  'orc.compress'='SNAPPY',
  'orc.create.index'='true'
);
